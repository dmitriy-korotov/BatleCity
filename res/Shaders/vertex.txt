#version 440
layout(location = 0) in vec3 vertex_position;
layout(location = 1) in vec3 vertex_color;
layout(location = 2) in vec2 texture_coords;
out vec3 color;
out vec2 tex_coords;
void main() 
{
   gl_Position = vec4(vertex_position, 1.0);
   tex_coords = texture_coords;
   color = vertex_color;
}
